54       /* 
55        * The function transferred accepts a request, and create the accounts 
56        * Object. We validate the accounts details by using validate() 
57        * method of the $request object based on th Rules: 
58        * SufficientFunds(), LimitFundWithdrawal() and DailylimitAmount() 
59        * If all conditions are met the code after the validation is 
60        * executed otherwise we return a response to the user with the 
61        * errors specified by the rules. 
62        * I demonstrate that in many transfer all the operation must 
63        * succeed otherwise all the operation reversed by implementing 
64        * the operations with a transaction. I do so by calling 
65        * DB::beginTransaction() and ending the transaction with a commit. 
66        * The user is then notified about the transfer of finds. 
67        * We do so by implementing the notify() function. An email is sent 
68        * to the user 
69        * */ 
70    
71       public  function transferred(Request $request) 
72       { 
73           $accountfrom=$request->input('accountfrom'); 
74           $accountto=$request->input('accountto'); 
75           $amount=$request->input('amount'); 
76    
77           $account=Account::find($accountfrom); 
78    
79           $request->validate([ 
80               'amount' => [ 
81                   'required', new SufficientFunds($accountfrom, 
82                           $accountto,$amount), 
83                   new LimitFundWithdrawal($accountfrom), 
84                   new DailyLimitAmount($accountfrom) 
85               ], 
86    
87           ]); 
88    
89    
90        if($account->balance<$amount) 
91        { 
92            $msg='You dont have sufficient amount in your account please add more'; 
93            return view('errorfundstransfer',compact(['msg','no'])); 
94        } 
95    
96    
97     $accountfrom1=  Account::find($accountfrom); 
98     $accountto1=Account::find($accountto); 
99     try { 
100        DB::beginTransaction(); 
101        $accountfrom1->balance -= $amount; 
102        $accountto1->balance += $amount; 
103        $accountfrom1->save(); 
104        $accountto1->save(); 
105   
106   
107        $trans1from = new Transaction; 
108        $transid = Transaction::all() 
109                ->max('transid') + 1; 
110        $trans1from->transid = $transid; 
111        $trans1from->transref = "Inter Account To " . $accountto; 
112        $trans1from->accountno = $accountfrom; 
113        $trans1from->amount = -1 * $amount; 
114        $trans1from->save(); 
115   
116        $trans2to = new Transaction; 
117        $transid = Transaction::all() 
118                ->max('transid') + 1; 
119        $trans2to->transid = $transid; 
120        $trans2to->transref = "Inter Account from " . $accountfrom; 
121        $trans2to->accountno = $accountto; 
122        $trans2to->amount = $amount; 
123        $trans2to->save(); 
124   
125        DB::commit(); 
126   
127       User::find(Auth::id())->notify(new NotifyIntraAccountTransfer($accountfrom1,$accountto1,$amount)); 
128   
129   
130        return \redirect('/'); 
131    }catch (Exception $ee) 
132    { 
133        DB::rollBack(); 
134        $err=$ee->getMessage(); 
135        return view('errorfundstransfer',compact('err')); 
136    }

194   public  function deposit(Request $request) 
195   { 
196       try { 
197           DB::beginTransaction(); 
198           $accountno = $request->input("accountfrom"); 
199           $transref = $request->input("transref"); 
200           $amount = $request->input("amount"); 
201           $account = Account::find($accountno); 
202           //  dd($account); 
203           $account->balance = $account->balance + $amount; 
204           $account->save(); 
205   
206           $transaction = new Transaction; 
207           $transaction->accountno = $accountno; 
208           $transaction->amount = $amount; 
209           $transaction->transref = $transref; 
210           $transid = Transaction::all() 
211                   ->max('transid') + 1; 
212           $transaction->transid = $transid; 
213           $transaction->save(); 
214   
215           DB::commit(); 
216           Notification::send(User::find(Auth::id()),new NotifyCashDeposited( $account,$amount)); 
217   
218       }catch(\Dompdf\Exception $e){ 
219           DB::rollback(); 
220   
221           Notification::send(User::find(Auth::id()),new NotifyCashDepositFailure($request)); 
222       } 
223   
224       return \redirect('/'); 
225
