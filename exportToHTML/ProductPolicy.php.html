7    namespace App\Policies;

9    use App\User; 
10   use App\Product; 
11   use Illuminate\Auth\Access\HandlesAuthorization; 
12    
13   class ProductPolicy 
14   { 
15       use HandlesAuthorization; 
16    
17       /* 
18        * A user can add a new product if he/she has administrator role 
19        */ 
20       public function store(User $user) 
21       { 
22           if($user->roles()->where('rolename','admin') 
23                   ->count()>0) 
24           { 
25               return true; 
26           }else{ 
27               return false; 
28           } 
29       } 
30    
31       /* 
32        * A user can delete a  product if he/she has an administrator role 
33       */ 
34       public function delete(User $user, Product $product) 
35       { 
36           if($user->roles()->where('rolename','admin') 
37                   ->count()>0) 
38           { 
39               return true; 
40           }else{ 
41               return false; 
42           } 
43       } 
44   }
